**1. Implement acceptSuggestion** in OverlayController:

Per RESEARCH.md Pattern 5 and CONTEXT.md D-09, D-10.

CRITICAL FIX (addresses review HIGH concern 2 -- repositionAfterAccept diverges from RESEARCH.md): The original plan used scalar offset arithmetic as the sole repositioning mechanism. RESEARCH.md explicitly recommends re-querying AX bounds as the authoritative source because target apps may normalize text on replacement (e.g., smart quotes, autocorrect, whitespace collapsing). The fix: use shifted scalar offsets ONLY for constructing CFRanges to pass to AX bounds re-query. The returned CGRects from AX are authoritative for positioning. If AX re-query fails for a suggestion, drop that suggestion from the display.

```swift
func acceptSuggestion(_ suggestion: Suggestion, context: TextContext, replacementOverride: String? = nil) {
    guard let replacement = replacementOverride ?? suggestion.primaryReplacement else { return }

    guard let offsetIndex = suggestions.firstIndex(where: { $0.id == suggestion.id }) else { return }
    let offset = suggestionScalarOffsets[offsetIndex]

    // 1. Try range-targeted AX write first (preferred: integrates with undo stack)
    let (settableErr, isSettable) = accessor.isAttributeSettable(
        context.axElement, kAXSelectedTextRangeAttribute
    )
    let supportsRangeWrite = settableErr == .success && isSettable

    var writeSucceeded = false

    if supportsRangeWrite {
        var cfRange = CFRange(location: offset.scalarStart, length: offset.scalarLength)
        guard let rangeValue = AXValueCreate(.cfRange, &cfRange) else { return }
        let selectError = accessor.setAttributeValue(
            context.axElement, kAXSelectedTextRangeAttribute, rangeValue
        )
        if selectError == .success {
            let replaceError = accessor.setAttributeValue(
                context.axElement, kAXSelectedTextAttribute, replacement as CFString
            )
            writeSucceeded = (replaceError == .success)
        }
    }

    // 2. Fallback: read full text, replace in memory, write back entire value
    if !writeSucceeded {
        writeSucceeded = fallbackFullWrite(
            offset: offset, replacement: replacement, context: context
        )
    }

    guard writeSucceeded else { return }  // Silent fail per UI-SPEC: underline stays

    // 3. Close popover, notify caller
    closePopover()
    onAcceptSuggestion?(suggestion)

    // 4. Remove accepted suggestion
    let replacementScalarCount = replacement.unicodeScalars.count
    suggestions.remove(at: offsetIndex)
    if offsetIndex < suggestionScalarOffsets.count {
        suggestionScalarOffsets.remove(at: offsetIndex)
    }

    // 5. If no suggestions remain, dismiss overlay
    if suggestions.isEmpty {
        dismiss()
        return
    }

    // 6. Reposition remaining suggestions
    repositionAfterAccept(
        acceptedStart: offset.scalarStart,
        originalLength: offset.scalarLength,
        replacementLength: replacementScalarCount,
        context: context
    )
}
```

**2. Implement fallbackFullWrite**:
```swift
@discardableResult
private func fallbackFullWrite(
    offset: (scalarStart: Int, scalarLength: Int),
    replacement: String,
    context: TextContext
) -> Bool {
    let (readError, readRef) = accessor.copyAttributeValue(context.axElement, kAXValueAttribute)
    guard readError == .success, let currentText = readRef as? String else { return false }

    let scalars = currentText.unicodeScalars
    let startIdx = scalars.index(scalars.startIndex, offsetBy: offset.scalarStart)
    let endIdx = scalars.index(startIdx, offsetBy: offset.scalarLength)
    guard let stringStart = startIdx.samePosition(in: currentText),
          let stringEnd = endIdx.samePosition(in: currentText) else { return false }

    var newText = currentText
    newText.replaceSubrange(stringStart..<stringEnd, with: replacement)

    let writeError = accessor.setAttributeValue(
        context.axElement, kAXValueAttribute, newText as CFString
    )
    return writeError == .success
}
```

**3. Implement repositionAfterAccept** (REVISED per review HIGH concern 2):

The critical insight from RESEARCH.md: scalar offset shifting is used to construct CFRanges for re-query, but the re-queried AX bounds are the AUTHORITATIVE source for underline positioning. If the target app modifies text in unexpected ways (smart quotes, autocorrect), the shifted offsets may be slightly wrong, but the AX bounds query will return the correct pixel position or nil.

```swift
private func repositionAfterAccept(
    acceptedStart: Int,
    originalLength: Int,
    replacementLength: Int,
    context: TextContext
) {
    let delta = replacementLength - originalLength

    // 1. Re-read the element's current text to get updated content
    let (valError, valRef) = accessor.copyAttributeValue(context.axElement, kAXValueAttribute)
    guard valError == .success, let newText = valRef as? String else {
        // Cannot verify text state -- dismiss to avoid stale positioning
        dismiss()
        return
    }

    // 2. Shift scalar offsets for suggestions AFTER the accepted range.
    //    These shifted offsets are used to construct CFRanges for AX bounds re-query.
    //    They are NOT used directly for positioning -- AX bounds are authoritative.
    for i in 0..<suggestionScalarOffsets.count {
        let off = suggestionScalarOffsets[i]
        // Check for overlap with accepted range (should not happen if Harper doesn't
        // produce overlapping ranges per Assumption A3, but handle defensively)
        let overlapStart = max(off.scalarStart, acceptedStart)
        let overlapEnd = min(off.scalarStart + off.scalarLength, acceptedStart + originalLength)
        if overlapStart < overlapEnd {
            // Overlapping suggestion is now invalid -- zero out so re-query fails gracefully
            suggestionScalarOffsets[i] = (scalarStart: off.scalarStart, scalarLength: 0)
            continue
        }
        if off.scalarStart >= acceptedStart + originalLength {
            suggestionScalarOffsets[i] = (
                scalarStart: off.scalarStart + delta,
                scalarLength: off.scalarLength
            )
        }
    }

    // 3. Update stored text context with new text
    textContext = TextContext(
        text: newText,
        bundleID: context.bundleID,
        extractionMethod: context.extractionMethod,
        selectionRange: context.selectionRange,
        elementBounds: context.elementBounds,
        axElement: context.axElement
    )

    // 4. Re-query AX bounds for ALL remaining suggestions using shifted offsets.
    //    AX bounds are AUTHORITATIVE -- if AX returns nil, drop that suggestion.
    //    This addresses review HIGH concern 2: re-queried bounds are the source of
    //    truth for positioning, not the shifted scalar offsets.
    guard suggestionScalarOffsets.count == suggestions.count else { return }

    var newEntries: [UnderlineEntry] = []
    var survivingSuggestions: [Suggestion] = []
    var survivingOffsets: [(scalarStart: Int, scalarLength: Int)] = []
    let windowOrigin = overlayWindow.frame.origin

    for (i, suggestion) in suggestions.enumerated() {
        // Use BoundsValidator for the re-query (handles multi-line, app quirks, watchdog)
        guard let rects = boundsValidator.validatedBoundsForRange(
            suggestion, in: newText, element: context.axElement,
            bundleID: context.bundleID, accessor: accessor
        ) else { continue }  // AX re-query failed -- drop this suggestion

        survivingSuggestions.append(suggestion)
        survivingOffsets.append(suggestionScalarOffsets[i])

        for rect in rects {
            let underlineRect = NSRect(x: rect.origin.x, y: rect.origin.y, width: rect.width, height: 2)
            let hitRect = UnderlineView.expandedHitRect(from: underlineRect)
            newEntries.append(UnderlineEntry(
                underlineRect: underlineRect.offsetBy(dx: -windowOrigin.x, dy: -windowOrigin.y),
                hitRect: hitRect.offsetBy(dx: -windowOrigin.x, dy: -windowOrigin.y),
                suggestion: suggestion
            ))
        }
    }

    suggestions = survivingSuggestions
    suggestionScalarOffsets = survivingOffsets

    // 5. If all re-queries failed, dismiss
    if newEntries.isEmpty {
        dismiss()
        return
    }

    // 6. Update underline view with new rects
    if let view = underlineView {
        view.entries = newEntries
        view.needsDisplay = true
    }
}
```

Note: `boundsValidator` is a `BoundsValidator` instance stored as a property on OverlayController. If this class does not exist yet at Plan 03 execution time, use the raw `boundsForRange` method from Plan 01 instead. BoundsValidator was introduced in Phase 6 as a refinement. The important principle is: AX bounds re-query is authoritative, scalar offsets are only for constructing the CFRange parameter.
