**1. Create TargetAppObserver** (OpenGram/SuggestionUI/TargetAppObserver.swift):

Per RESEARCH.md Pattern 6.

CRITICAL FIX (addresses review HIGH concern 1 -- Unmanaged pointer bug): The original plan called `Unmanaged.passRetained(context)` twice -- once in `install()` and once in `uninstall()`. The second `passRetained` creates a NEW retain, meaning the original retain from `install()` is never released (leak), and the `uninstall()` retain is released immediately (use-after-free risk). The fix: store the `Unmanaged` reference from the initial `passRetained` as a property, and call `.release()` on that stored reference in `uninstall()`.

```swift
@preconcurrency import ApplicationServices
import Foundation

final class TargetAppObserver {
    private var axObserver: AXObserver?
    private var dismissContext: DismissContext?
    /// Stores the Unmanaged reference from passRetained so we release the SAME
    /// retain we created -- never call passRetained a second time.
    private var unmanagedContext: Unmanaged<DismissContext>?

    func install(pid: pid_t, onDismiss: @escaping @MainActor () -> Void) {
        uninstall()  // clean up any prior observer

        let context = DismissContext(handler: onDismiss)
        self.dismissContext = context
        // passRetained increments the retain count by 1. Store the Unmanaged
        // wrapper so uninstall() can call release() on the exact same retain.
        let unmanaged = Unmanaged.passRetained(context)
        self.unmanagedContext = unmanaged
        let ptr = unmanaged.toOpaque()

        var observer: AXObserver?
        AXObserverCreate(pid, { _, _, _, userData in
            guard let userData else { return }
            let ctx = Unmanaged<DismissContext>.fromOpaque(userData).takeUnretainedValue()
            // Addresses review MEDIUM concern 7: use Task { @MainActor in } instead of
            // MainActor.assumeIsolated, which is unsafe from a non-main-actor context.
            Task { @MainActor in
                ctx.handler()
            }
        }, &observer)

        guard let observer else { return }

        let appElement = AXUIElementCreateApplication(pid)
        let notifications: [String] = [
            kAXWindowMovedNotification,
            kAXWindowResizedNotification,
            kAXFocusedUIElementChangedNotification,
            kAXApplicationDeactivatedNotification,
        ]
        for notification in notifications {
            AXObserverAddNotification(observer, appElement, notification as CFString, ptr)
        }

        CFRunLoopAddSource(
            RunLoop.main.getCFRunLoop(),
            AXObserverGetRunLoopSource(observer),
            .defaultMode
        )
        self.axObserver = observer
    }

    func uninstall() {
        if let observer = axObserver {
            CFRunLoopRemoveSource(
                RunLoop.main.getCFRunLoop(),
                AXObserverGetRunLoopSource(observer),
                .defaultMode
            )
            axObserver = nil
        }
        // Release the STORED Unmanaged reference -- balances the single passRetained in install().
        // Do NOT call Unmanaged.passRetained(context).release() here -- that would create
        // a NEW retain and immediately release it, leaving the original retain leaked.
        if let unmanaged = unmanagedContext {
            unmanaged.release()
            unmanagedContext = nil
        }
        dismissContext = nil
    }

    deinit {
        // Safety net: clean up if caller forgot to uninstall
        if axObserver != nil {
            uninstall()
        }
    }
}

// @unchecked Sendable: handler is always dispatched onto the main actor via Task { @MainActor in },
// so concurrent access to the stored closure is not possible at runtime.
private final class DismissContext: @unchecked Sendable {
    let handler: @MainActor () -> Void

    init(handler: @escaping @MainActor () -> Void) {
        self.handler = handler
    }
}
```
